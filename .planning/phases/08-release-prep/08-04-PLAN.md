   ```json
   {

     "version": "0.0.0",
     "private": true,
     "description": "Demo of tinyimg-unplugin with Webpack",
     "scripts": {
       "build": "webpack --mode production"
     },
     "devDependencies": {

       "webpack-cli": "^5.0.0",
       "tinyimg-unplugin": "^0.1.0-beta.1"
     }
   }
   ```

3. Create webpack.config.js:
   ```javascript
   const path = require('path')
   const { tinyimgUnplugin } = require('tinyimg-unplugin')

   module.exports = {
     entry: './src/index.js',
     output: {
       path: path.resolve(__dirname, 'dist'),
       filename: 'bundle.js'
     },
     plugins: [
       tinyimgUnplugin.webpack({
         mode: 'random',
         cache: true,
         parallel: 8
       })
     ]
   }
   ```

4. Create src/index.js:
   ```javascript
   import largeImage from './assets/large-image.png'

   const img = document.createElement('img')
   img.src = largeImage
   document.body.appendChild(img)

   console.log('Image loaded:', largeImage)
   ```

5. Create index.html:
   ```html
   <!DOCTYPE html>

     <meta charset="UTF-8">
     <title>tinyimg-unplugin Webpack Demo</title>
   </head>
   <body>
     <h1>tinyimg-unplugin + Webpack Demo</h1>
     <script src="dist/bundle.js"></script>
   </body>
   </html>
   ```
